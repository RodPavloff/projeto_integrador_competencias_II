 20def create_user(username, email, password):
 21    # Conectar ao banco de dados MySQL
 22    conn = mysql.connector.connect(
 23        host='localhost',
 24        user='root',
 25        password='root',
 26        database='PIT_II'
 27    )
 28
 29    try:
 30        with conn.cursor() as cursor:
 31            # Verificar se o usuário já existe
 32            cursor.execute('SELECT id_cliente FROM cliente WHERE nome=%s', (username,))
 33            existing_user = cursor.fetchone()
 34
 35            if existing_user:
 36                st.error("Usuário já existe. Por favor, escolha outro nome.")
 37                return False
 38
 39            # Criar um novo usuário
 40            hashed_password = hl.sha256(password.encode()).hexdigest()
 41            cursor.execute('INSERT INTO cliente (nome, email, senha) VALUES (%s, %s, %s)', (username, email, hashed_password))
 42            st.success(f"Usuário '{username}' criado com sucesso!")
 43
 44    except mysql.connector.Error as err:
 45        st.error(f"Erro MySQL: {err}")
 46        return False
 47
 48    finally:
 49        # Commit e fechar a conexão
 50        conn.commit()
 51        conn.close()
 52
 53    return True

 55def authenticate_user(username, password):
 56    # Conectar ao banco de dados MySQL
 57    conn = mysql.connector.connect(
 58        host='localhost',
 59        user='root',
 60        password='root',
 61        database='PIT_II'
 62    )
 63
 64    try:
 65        with conn.cursor() as cursor:
 66            # Buscar usuário por nome
 67            cursor.execute('SELECT id_cliente, senha FROM cliente WHERE nome=%s', (username,))
 68            user_data = cursor.fetchone()
 69
 70            if not user_data:
 71                # Remova esta linha para permitir login para usuários não cadastrados
 72                # st.error("Usuário não encontrado. Por favor, verifique o nome.")
 73                pass
 74
 75            # Verificar senha
 76            hashed_password = hl.sha256(password.encode()).hexdigest()
 77            if hashed_password != user_data[1]:
 78                st.error("Senha incorreta. Por favor, tente novamente.")
 79                return False
 80
 81            st.success(f"Login bem-sucedido! Bem-vindo, {username}!")
 82
 83    except mysql.connector.Error as err:
 84        st.error(f"Erro MySQL: {err}")
 85        return False
 86
 87    finally:
 88        # Fechar a conexão
 89        conn.close()
 90
 91    return True
